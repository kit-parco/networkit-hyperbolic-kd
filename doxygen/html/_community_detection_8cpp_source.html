252 Graph getGraph(OptionParser::Option* options) {
253 
254  // generated graph
255  if (options[GENERATE]) {
256  std::string genOption = options[GENERATE].arg;
257  std::cout << "\t --generate=" << options[GENERATE].arg << std::endl;
258 
259  // get graph generation model
260  std::vector<std::string> genArgs = splitAString(genOption, ':');
261  std::vector<std::string> genNumArgs = splitAString(genArgs.at(1), ',');
262  assert (genArgs.size() == 2);
263  std::string model = genArgs.at(0);
264  if (model == "RG") { // random graph (Erdos-Renyi)
265  assert (genNumArgs.size() == 2);
266  count n = std::atoi(genNumArgs.at(0).c_str());
267  double p = std::atoi(genNumArgs.at(1).c_str());
268  return generateRandomGraph(n, p);
269  } else if (model == "CR") { // clustered random graph
270  assert (genNumArgs.size() == 4);
271  count n = std::atoi(genNumArgs.at(0).c_str());
272  count k = std::atoi(genNumArgs.at(1).c_str());
273  double pin = std::atof(genNumArgs.at(2).c_str());
274  double pout = std::atof(genNumArgs.at(3).c_str());
275  return generateClusteredRandomGraph(n, k, pin, pout);
276  } else if (model == "PA") { // preferential attachment (Barabasi-Albert)
277  assert (genNumArgs.size() == 2);
278  count n = std::atoi(genNumArgs.at(0).c_str());
279  count a = std::atoi(genNumArgs.at(1).c_str());
280  return generatePreferentialAttachmentGraph(n, a);
281  } else {
282  std::cout << "[ERROR] unknown graph generation model: " << model << " [EXIT]" << std::endl;
283  exit(1);
284  }
285 
286 
287  } else if (options[GRAPH]) { // graph from file
288  std::string graphPath = options[GRAPH].arg;
289  std::cout << "\t --graph=" << graphPath << std::endl;
290 
291  Graph G = readGraph(graphPath);
292  return G;
293  } else {
294  Graph G(0); // return empty graph
295  G.setName("NONE");
296  return G;
297  }

402  } else {

505 
506  std::cout << "[INFO] Graph: " << G.toString() << std::endl;
507

522

525  }
526 
527

529  std::cout << "\t modularity:\t" << mod << std::endl;
530  std::cout << std::endl; // newline

532 
533  if (options[SUMMARY]) {
534  char sep = ';';
535  std::ofstream summary(options[SUMMARY].arg, std::ios::app); // open summary file to append to
536  summary << k << sep;
537  summary << mod << std::endl; // since this is the last column, append endl instead of separator
538  }

32 Clustering EnsemblePreprocessing::run(Graph& G) {
33  INFO("STARTING EnsemblePreprocessing on G=" << G.toString());
34 
35  // fixed sub-algorithms
36  ClusterContracter contracter;
37  ClusteringProjector projector;
38 
39  // data
40  std::vector<Clustering> baseClusterings(baseClusterers.size(), Clustering(0)); // collection of base clusterings - fill with empty clustering
41 
42  // run base clusterers in parallel
43  #pragma omp parallel for
44  for (int b = 0; b < baseClusterers.size(); b += 1) {
45  baseClusterings.at(b) = baseClusterers.at(b)->run(G);
46  }
47 
48  // ANALYSIS
49  if (CALC_DISSIMILARITY) {
50  JaccardMeasure dm;
51  double dissimilaritySum = 0.0;
52  for (int b = 0; b < baseClusterings.size(); b += 1) {
53  for (int c = b + 1; c < baseClusterings.size(); c += 1) {
54  double d = dm.getDissimilarity(G, baseClusterings.at(b), baseClusterings.at(c));
55  dissimilaritySum += d;
56  }
57  }
58  double avgDissimilarity = dissimilaritySum / (baseClusterings.size() * (baseClusterings.size() - 1) / 2.0);
59  std::cout << "[INFO] avg. base clustering dissimilarity: " << avgDissimilarity << std::endl;
60  }
61  //
62 
63  // create core clustering
64  Clustering core = this->overlap->run(G, baseClusterings);
65  // contract graph according to core clustering
66  std::pair<Graph, NodeMap<node> > contraction = contracter.run(G, core);
67  Graph Gcore = contraction.first;
68  NodeMap<node> fineToCoarse = contraction.second;
69  // send contracted graph to final clusterer
70  Clustering finalCoarse = this->finalClusterer->run(Gcore);
71 
72  // project clustering of contracted graph back to original graph
73  Clustering final = projector.projectBack(Gcore, G, fineToCoarse, finalCoarse);
74  // return clustering
75  return final;
76 }

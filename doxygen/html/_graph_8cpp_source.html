45 void Graph::addEdge(node u, node v, edgeweight weight) {
46  assert (u >= 0);
47  assert (u <= this->z);
48  assert (v >= 0);
49  assert (v <= this->z); // node ids must be in range
50 
51  if (u == v) { // self-loop case
52  this->adja[u].push_back(u);
53  this->deg[u] += 1;
54  this->eweights[u].push_back(weight);
55  for (index attrId = 0; attrId < this->edgeMaps_double.size(); ++attrId) {
56  double defaultAttr = this->edgeAttrDefaults_double[attrId];
57  this->edgeMaps_double[attrId][u].push_back(defaultAttr);
58  }
59  } else {
60  // set adjacency
61  this->adja[u].push_back(v);
62  this->adja[v].push_back(u);
63  // increment degree counters
64  this->deg[u] += 1;
65  this->deg[v] += 1;
66  // set edge weight
67  this->eweights[u].push_back(weight);
68  this->eweights[v].push_back(weight);
69  // loop over all attributes, setting default attr
70  for (index attrId = 0; attrId < this->edgeMaps_double.size(); ++attrId) {
71  double defaultAttr = this->edgeAttrDefaults_double[attrId];
72  this->edgeMaps_double[attrId][u].push_back(defaultAttr);
73  this->edgeMaps_double[attrId][v].push_back(defaultAttr);
74  }
75  }
76 
77  m++; // increasing the number of edges
78 }

372 node Graph::mergeEdge(node u, node v, bool discardSelfLoop) {
373  DEBUG("merge edge with nodes " << u << " and " << v);
374 
375  if (u != v) {
376  node newNode = this->addNode();
377 
378  // self-loop if necessary
379  if (! discardSelfLoop) {
380  TRACE("selfLoopWeight");
381  edgeweight selfLoopWeight = this->weight(u, u) + this->weight(v, v) + this->weight(u, v);
382  this->addEdge(newNode, newNode, selfLoopWeight);
383  TRACE("end selfLoopWeight");
384  }
385 
386  // rewire edges from u to newNode
387  this->forWeightedEdgesOf(u, [&](node u, node neighbor, edgeweight w) {
388  if (neighbor != u) {
389  TRACE("neighbor of " << u << ": " << neighbor);
390  this->addEdge(neighbor, newNode, this->weight(u, neighbor)); // TODO: make faster
391  TRACE("end neighbor of u");
392  }
393  });
394 
395  // rewire edges from v to newNode
396  this->forWeightedEdgesOf(v, [&](node v, node neighbor, edgeweight w) {
397  if (neighbor != v) {
398  TRACE("neighbor of " << v << ": " << neighbor);
399  this->addEdge(neighbor, newNode, this->weight(v, neighbor)); // TODO: make faster
400  TRACE("end neighbor of v");
401  }
402  });
403 
404  // delete edges of nodes to delete
405  this->forEdgesOf(u, [&](node u, node neighbor) {
406  this->removeEdge(u, neighbor);
407  });
408  this->forEdgesOf(v, [&](node v, node neighbor) {
409  this->removeEdge(v, neighbor);
410  });
411 
412  // delete nodes
413  this->removeNode(u);
414  this->removeNode(v);
415 
416  return newNode;
417  }
418 
419  // no new node created
420  return none;
421 }

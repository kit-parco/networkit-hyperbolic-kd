645 inline void NetworKit::Graph::breadthFirstNodesFrom(node r, std::vector<int>& marked, L handle) {
646  DEBUG("Call to BFS");
647  std::queue<node> q;
648  count n = this->numberOfNodes();
649  q.push(r); // enqueue root
650  marked[r] = 1;
651  do {
652  node u = q.front();
653  q.pop();
654  // apply function
655  handle(u);
656  this->forNeighborsOf(u, [&](node v) {
657  // filtering edges is not necessary because only out-edges are considered by stinger
658  if (marked[v] == 0) {
659  q.push(v);
660  marked[v] = 1;
661  }
662  });
663  } while (!q.empty());
664 }

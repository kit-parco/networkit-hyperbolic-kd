22 Clustering Louvain::pass(Graph& G) {
23 
24  // init clustering to singletons
25  count n = G.numberOfNodes();
26  Clustering zeta(n);
27  zeta.allToSingletons();
28 
29  // $\omega(E)$
30  edgeweight total = G.totalEdgeWeight();
31 
32  // For each node we store a map that maps from cluster ID
33  // to weight of edges to that cluster, this needs to be updated when a change occurs
34  std::vector<std::map<cluster, edgeweight> > incidenceWeight(
35  G.numberOfNodes());
36  G.parallelForNodes([&](node u) {
37  G.forWeightedEdgesOf(u, [&](node u, node v, edgeweight w) {
38  cluster C = zeta[v];
39  if (u != v) {
40  incidenceWeight[u][C] += w;
41  }
42  });
43  });
44 
45 #ifdef _OPENMP
46  // create and init locks
47  std::vector<omp_lock_t> mapLocks(n);
48  G.parallelForNodes([&](node u) {
49  omp_init_lock(&mapLocks[u]);
50  });
51 #endif
52 
53  // modularity update formula for node moves
54  // $$\Delta mod(u:\ C\to D)=\frac{\omega(u|D)-\omega(u|C\setminus v)}{\omega(E)}+\frac{2\cdot\vol(C\setminus u)\cdot\vol(u)-2\cdot\vol(D)\cdot\vol(u)}{4\cdot\omega(E)^{2}}$$
55 
56  // parts of formula follow
57  NodeMap<double> volNode(G.numberOfNodes(), 0.0);
58  // calculate and store volume of each node
59  G.parallelForNodes([&](node u) {
60  volNode[u] += G.weightedDegree(u);
61  volNode[u] += G.weight(u, u); // consider self-loop twice
62  });
63 
64  IndexMap<cluster, double> volCluster(G.numberOfNodes(), 0.0);
65  // set volume for all singletons
66  zeta.parallelForEntries([&](node u, cluster C) {
67  volCluster[C] = volNode[u];
68  });
69  // end of initialization, set barrier for safety reasons
70 
71  // $\vol(C \ {x})$ - volume of cluster C excluding node x
72  auto volClusterMinusNode = [&](cluster C, node x) {
73  double volC = 0.0;
74  double volN = 0.0;
75 #pragma omp atomic read
76  volC = volCluster[C];
77  if (zeta[x] == C) {
78 #pragma omp atomic read
79  volN = volNode[x];
80  return volC - volN;
81  } else {
82  return volC;
83  }
84  };
85 
86  // $\omega(u | C \ u)$
87  auto omegaCut = [&](node u, cluster C) {
88  edgeweight w = 0.0;
89 #ifdef _OPENMP
90  omp_set_lock(&mapLocks[u]);
91 #endif
92  // #pragma omp atomic read
93  w = incidenceWeight[u][C];
94 #ifdef _OPENMP
95  omp_unset_lock(&mapLocks[u]);
96 #endif
97  return w;
98  };
99 
100  // difference in modularity when moving node u from cluster C to D
101  auto deltaMod =
102  [&](node u, cluster C, cluster D) {
103  double volN = 0.0;
104 #pragma omp atomic read
105  volN = volNode[u];
106  double delta = (omegaCut(u, D) - omegaCut(u, C)) / total + this->gamma * ((volClusterMinusNode(C, u) - volClusterMinusNode(D, u)) * volN) / (2 * total * total);
107  return delta;
108  };
109 
110 
111  // begin pass
112  int i = 0;
113  bool change = false; // change in last iteration?
114  do {
115  i += 1;
116  DEBUG("---> Louvain pass: iteration # " << i << ", parallelism: " << this->parallelism);
117  change = false; // is clustering stable?
118 
119  // try to improve modularity by moving a node to neighboring clusters
120  auto moveNode = [&](node u) {
121 // std::cout << u << " ";
122 
123  cluster best = none;
124  cluster C = none;
125  cluster D = none;
126  double deltaBest = -0.5;
127 
128 #pragma omp atomic read
129  C = zeta[u];
130 
131 // TRACE("Processing neighborhood of node " << u << ", which is in cluster " << C);
132  G.forNeighborsOf(u, [&](node v) {
133 #pragma omp atomic read
134  D = zeta[v];
135 // TRACE("Neighbor " << v << ", which is still in cluster " << zeta[v]);
136  if (D != C) { // consider only nodes in other clusters (and implicitly only nodes other than u)
137  double delta = deltaMod(u, C, D);
138  if (delta > deltaBest) {
139  deltaBest = delta;
140  best = D;
141  }
142  }
143  });
144 
145  if (deltaBest > 0.0) { // if modularity improvement possible
146  assert (best != C && best != none);// do not "move" to original cluster
147  double volN = 0.0;
148 
149  // update weight of edges to incident clusters
150  G.forWeightedNeighborsOf(u, [&](node v, edgeweight w) {
151 #ifdef _OPENMP
152  omp_set_lock(&mapLocks[v]);
153 #endif
154  incidenceWeight[v][best] += w;
155  incidenceWeight[v][C] -= w;
156 #ifdef _OPENMP
157  omp_unset_lock(&mapLocks[v]);
158 #endif
159  });
160 #pragma omp atomic write
161  zeta[u] = best; // move to best cluster
162 #pragma omp atomic read
163  volN = volNode[u];
164 #pragma omp atomic write
165  change = true; // change to clustering has been made
166 #pragma omp atomic write
167  this->anyChange = true; // indicate globally that clustering was modified
168 
169  // update the volume of the two clusters
170 #pragma omp atomic update
171  volCluster[C] -= volN;
172 #pragma omp atomic update
173  volCluster[best] += volN;
174  }
175  };
176 
177  // apply node movement according to parallelization strategy
178  if (this->parallelism == "none") {
179  G.forNodes(moveNode);
180  } else if (this->parallelism == "simple") {
181  G.parallelForNodes(moveNode);
182  } else if (this->parallelism == "balanced") {
183  G.balancedParallelForNodes(moveNode);
184  } else {
185  ERROR("unknown parallelization strategy: " << this->parallelism);
186  exit(1);
187  }
188 
189 // std::cout << std::endl;
190  } while (change && i < MAX_LOUVAIN_ITERATIONS);
191 
192  // free lock mem
193 #ifdef _OPENMP
194 #pragma omp barrier
195  DEBUG("about to destroy locks in Louvain pass");
196  G.parallelForNodes([&](node u) {
197  omp_destroy_lock(&mapLocks[u]);
198  });
199 #endif
200 
201  return zeta;
202 }

204 Clustering Louvain::run(Graph& G) {
205  INFO("starting Louvain method");
206 
207  // sub-algorithms
208  ClusterContracter contracter;
209  ClusteringProjector projector;
210 
211  // hierarchies
212  std::vector<std::pair<Graph, NodeMap<node> > > hierarchy; // hierarchy of graphs G^{i} and maps M^{i->i+1}
213  std::vector<NodeMap<node> > maps; // hierarchy of maps M^{i->i+1}
214 
215  int h = -1; // finest hierarchy level
216  bool done = false; //
217 
218  Graph* graph = &G;
219 
220  do {
221  h += 1; // begin new hierarchy level
222  DEBUG("Louvain hierarchy level " << h);
223  // one local optimization pass
224  DEBUG("starting Louvain pass");
225  Clustering clustering = this->pass(*graph);
226  if (this->anyChange) {
227  // contract the graph according to clustering
228  DEBUG("starting contraction");
229  hierarchy.push_back(contracter.run(*graph, clustering));
230  maps.push_back(hierarchy[h].second);
231  graph = &hierarchy[h].first;
232 
233  done = false; // new hierarchy level, continue with loop
234  this->anyChange = false; // reset change flag
235  } else {
236  done = true; // if clustering was not modified, do not contract and exit loop
237  }
238  } while (!done);
239 
240  DEBUG("starting projection");
241  // project fine graph to result clustering
242  Clustering result = projector.projectCoarseGraphToFinestClustering(*graph,
243  G, maps);
244  return result;
245 }
